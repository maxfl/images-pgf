\documentclass[png]{standalone}

\usepackage{tikz}
\usepackage{tikz-relcoord}

\begin{document}
\begin{tikzpicture}
  \pgfdeclareimage[width=10mm]{virus}{./coronavirus.png};
  \draw node (bkg) {\includegraphics{./AlexNoriega_destaque.png}};

  % Simple positions
  \draw (bkg.north west) node[below right] {\pgfuseimage{virus}};
  \draw (bkg.north east) node[below left]  {\pgfuseimage{virus}};
  \draw (bkg.south west) node[above right] {\pgfuseimage{virus}};
  \draw (bkg.south east) node[above left]  {\pgfuseimage{virus}};

  % Using relative position:
  % \RelPos(<node>)(rel-x,rel-y)
  \draw \RelPos(bkg)(0.485,0.5) node {\pgfuseimage{virus}};

  % Define named coordinates
  % \RelCoord (coordname) at (rel-x,rel-y) of (<node>);
  \RelCoord (c1) at (0.15,0.8) of (bkg);
  \draw (c1) node {\pgfuseimage{virus}};

  % 30 random coordinates
  \foreach \coord in {1,2,...,30}{%
    \RelCoord (c) at (0.5+0.45*rand,0.5+0.45*rand) of (bkg);
    \draw (c) node {\pgfuseimage{virus}};
  }
\end{tikzpicture}
\end{document}
